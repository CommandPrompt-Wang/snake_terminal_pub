\section{项目概述}
% ============================================================

\begin{frame}{项目简介}
  \begin{itemize}
    \item \textbf{Snake Terminal}：C++17 + raylib 的双人图形化贪吃蛇
    \item 架构：Scene 回调式单线程主循环（参考 Godot / Unity 场景思路）
    \item 玩法可配置；含动画状态机、虚影复活、自研音频与日志
    \item 构建：CMake + \texttt{build.sh} / \texttt{build.bat}；配置：\texttt{snake.cfg}
    \item 依赖：raylib（子模块）、miniaudio、dr\_mp3
  \end{itemize}

  \vspace{0.4em}
  \begin{block}{目标}
    逻辑与表现解耦、场景可扩展，在双人玩法之上做出完整工程闭环。
  \end{block}
\end{frame}

\begin{frame}{场景流与生命周期}

      \begin{center}
        菜单 \\[0.2em]
        $\swarrow$\quad$\searrow$ \\[0.2em]

        \bottomrule
      \end{tabular}
    \end{column}
  \end{columns}

\begin{frame}{主循环（SceneManager::run）}
  \textbf{单线程}，每帧：
  \begin{enumerate}
    \item 按键白名单 $\rightarrow$ 构造 \texttt{InputEvent}（Press / Release / repeat）
    \item \texttt{handle\_event}：\texttt{on\_event} 通用钩子 $\rightarrow$ \texttt{dynamic\_cast} 分发
    \item \texttt{update(dt)}；若 \texttt{is\_finished} $\rightarrow$ \texttt{switch\_to}
    \item \texttt{BeginDrawing} $\rightarrow$ \texttt{render()} $\rightarrow$ \texttt{EndDrawing}
  \end{enumerate}

  \vspace{0.3em}
  \begin{alertblock}{单线程约束}
    所有 raylib 调用（InitWindow / LoadTexture / Draw*）必须同线程，
    否则 OpenGL context 与 \texttt{thread\_local} 出问题。
  \end{alertblock}
\end{frame}

\begin{frame}[fragile]{事件分发}
\begin{verbatim}
void handle_event(Event& event) {
  if (event.is_consumed()) return;
  on_event(event);         // (1) 通用钩子，可拦截
  if (event.is_consumed()) return;
  on_event_internal(event);// (2) dynamic_cast -> 具体类型
}
\end{verbatim}
  \vspace{0.3em}
  子类只需覆盖需要的处理程序（如 \texttt{on\_inputevent}），
  用 \texttt{event.consume()} 阻止后续处理。
\end{frame}

\begin{frame}{全局共享状态}
  \footnotesize
  \begin{itemize}
    \item 分数：\texttt{last\_score\_player1/2}
    \item 玩家状态：\texttt{ALIVE / ON\_WALL / ON\_SELF / ON\_PLAYER / STARVED}
    \item 结束原因（高位不覆盖低位）：\\
      \texttt{FULL\_BOARD > TIMEOUT > MANUAL > DEATH}
    \item 模式：\texttt{DEATHMATCH} / \texttt{TIMERACE}
  \end{itemize}

  \vspace{0.3em}
  结算显示：
  \begin{itemize}
    \item \texttt{FULL\_BOARD} $\rightarrow$ You've filled everywhere!!!
    \item \texttt{TIMEOUT} $\rightarrow$ TIME LIMIT REACHED
    \item \texttt{MANUAL} $\rightarrow$ GAME ENDED
    \item \texttt{DEATH} $\rightarrow$ 逐玩家显示死因（跳过 ALIVE）
  \end{itemize}
\end{frame}

\section{核心实现}
% ============================================================

\begin{frame}{蛇逻辑与动画分离}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Snake}（纯逻辑）
      \begin{itemize}
        \item 移动、碰撞、得分
        \item 复活、尾部切除
        \item 虚影 / 可碰撞开关
        \item \textbf{不感知}动画
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{SnakeBody} + AnimateManager
      \begin{itemize}
        \item \texttt{SnakeMove}：移动/加速
        \item \texttt{SnakeDie}：逐节 X
        \item \texttt{SnakeWaiting}：虚影/冻结
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.5em}
  由 \texttt{GameScene} 驱动转换：\\
  MOVE $\xrightarrow{\text{死亡}}$ DIE $\xrightarrow{\text{播完/打断}}$ WAITING $\xrightarrow{\text{按键}}$ MOVE
\end{frame}

\begin{frame}{虚影复活与动画打断}
  \textbf{Ghost}（TIMERACE + \texttt{respawnInAdvance}）：
  \begin{enumerate}
    \item 死亡 $\rightarrow$ 原位播放 X 标记动画
    \item 动画完成 $\rightarrow$ 先切尾扣分，再 \texttt{generate\_ghost()}（半透明）
    \item 玩家确认位置后按键 $\rightarrow$ \texttt{deploy\_from\_ghost()}（不再重复扣分）
  \end{enumerate}

  \vspace{0.3em}
  \textbf{死亡动画打断}（\texttt{deathAnimInterruptThreshold}）：
  \begin{itemize}
    \item $-1$：关闭，必须看完
    \item $0+$：前 N 节正常播，之后按移动键跳过剩余
  \end{itemize}
\end{frame}

\begin{frame}{音频系统}
  \textbf{共享设备 + 预解码 PCM + 回调混音}

  \begin{itemize}
    \item \texttt{AudioManager}：唯一 \texttt{ma\_device}（48kHz 立体声 f32）
    \item 模板库不直接播；\texttt{play\_sound} 独占，\texttt{play\_sfx} 克隆入并发池
    \item \texttt{AudioLoader}：加载时预解码 MP3 $\rightarrow$ float32 PCM
    \item \texttt{mix()}：线性插值重采样 + 音量 + 声道转换
    \item 设备持续运行，无声输出静音，避免 \texttt{ma\_device\_stop} 死锁
  \end{itemize}

  \vspace{0.2em}
  对比 raylib Music：主线程卡顿会导致欠载；本方案音频线程独立于帧率。
\end{frame}

\begin{frame}{日志系统}
  线程安全彩色日志，级别过滤 + 接管 raylib 日志：

  \begin{table}
    \centering
    \footnotesize
    \begin{tabular}{@{}lll@{}}
      \toprule
      级别 & 函数 & 颜色 \\
      \midrule
      Debug & \texttt{logd} & 青 \\
      Info & \texttt{log} & 绿 \\
      Warning & \texttt{logw} & 黄 \\
      Error & \texttt{loge} & 红 \\
      \bottomrule
    \end{tabular}
  \end{table}

  \vspace{0.3em}
  CLI：\texttt{./dist/snake --loglevel debug}（默认 Info）\\
  启动时切到可执行文件目录；Windows 额外设 UTF-8 + 禁用 IME。
\end{frame}

\section{设计取舍与贡献}
% ============================================================

\begin{frame}{已知限制与取舍}
  \footnotesize
  \begin{enumerate}
    \item \textbf{纹理无引用计数}（raylib）：各 Sprite 独享 texture，用完即卸，安全但费显存
    \item \textbf{自研音频}：绕开 raylib Music 主线程依赖；缺总线级实时混音，预用 loudnorm 平衡
    \item \textbf{Sprite 职责偏多}：加载 / 纹理 / 变换 / 帧动画合一；本规模下拆分收益不大
    \item \textbf{状态机持裸指针}：move 后需修复；双角色不上完整 ECS
    \item \textbf{物理帧=渲染帧}：无独立固定步；碰撞嵌在 Snake 友元函数中
  \end{enumerate}
\end{frame}

\begin{frame}{上游贡献}
  \begin{block}{raylib \texttt{ImageFromImage} 边界错误}
    矩形校验用了严格比较 $>$ / $<$ 而非 $\ge$ / $\le$，
    导致从 $(0,0)$ 裁剪或裁剪完整图像被错误拒绝。
  \end{block}

  \vspace{0.5em}
  已提交并合并：\\
  \url{https://github.com/raysan5/raylib/pull/5979}
\end{frame}
